Element Selector

div{
color: blue;
}

Class Selector

//highlight is the class in p element
.highlight {
color: green;
background-color: yellow;
}

ID Selector

//abcd is the id in p element
#abcd {
color: red;
background-color: yellowgreen;
}

Child Selector

//p is a child of div
div > h2 {         //div ko direct vitra vako h2 lai select garxa
color: white;
background-color: darkblue;
}

Descendent Selector

//p is a descendant of div
div p {         //div ko vitra jata vaye pani vako p lai select garxa
color: white;
background-color: darkblue;
}
eg: ABCD

Universal Selector

* {         //selects all elements
margin: 0;
padding: 0;
}

Pseudo Selector

p:hover {
color: darkorange;
}
